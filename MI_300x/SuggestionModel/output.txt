OPENAI_API_GPU_UTIL=0.9

===== ./compose.yml =====
services:
  suggestion_model:
      container_name: "suggestion-model"
      image: vllm/vllm-openai-rocm:latest
      volumes:
        - ./deployment/vllm/models:/root/.cache/huggingface
        - ./deployment/vllm/templates/:/app/templates/
      ports:
        - "${OPENAI_API_PORT}:${OPENAI_API_PORT}"
      ipc: host
      shm_size: 10g
      devices:
        - /dev/kfd
        - /dev/dri
      group_add:
        - video
      security_opt:
        - seccomp:unconfined
      cap_add:
        - SYS_PTRACE
      environment:
        - HIP_VISIBLE_DEVICES=0,1
      command: >
        --model ${OPENAI_API_MODEL}
        --port ${OPENAI_API_PORT}
        --gpu-memory-utilization ${OPENAI_API_GPU_UTIL}
        --api-key ${OPENAI_API_KEY}
        -tp 2
        --enable-auto-tool-choice
        --tool-call-parser openai


===== ./deployment.yml =====
---
- name: Install Docker and prepare all nodes
  hosts: amd_gpu_server
  become: false
  vars_files:
    - vault/secrets.yml
  vars:
    project_dir: "/home/{{ amd_gpu_server_user }}/artifactRegistory/SuggestionModel"
    ansible_user: "{{ amd_gpu_server_user }}"
    ansible_password: "{{ amd_gpu_server_password }}"

  tasks:
    - name: Create project directory
      file:
        path: "{{ project_dir }}"
        state: directory
        mode: "0755"

    - name: Copy docker compose file
      copy:
        src: "compose.yml"
        dest: "{{ project_dir }}/docker-compose.yml"

    - name: Copy environment file
      copy:
        src: ".env"
        dest: "{{ project_dir }}/.env"

    - name: Start Suggestion Model
      community.docker.docker_compose_v2:
        project_src: "{{ project_dir }}"
        state: present
        remove_orphans: true
